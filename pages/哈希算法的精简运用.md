---
title: 哈希算法的精简运用
---

```md
const array = [
    {
        time: '2018-11-14',
        text: '123'
    },
    {
        time: '2018-11-13',
        text: '321'
    },
    {
        time: '2018-11-28',
        text: '233'
    },
    {
        time: '2018-11-13',
        text: '322'
    }
]; //假设有这么一段对象数组格式

let data = {};
array.forEach( function(item, index) {
    if (!data[item.time]) {
        data[item.time] = [item.text]
    } else {
        data[item.time].push(item.text)
    }
})
console.log(data);
```
